% ---------- 章节标题（带装饰：section 左竖条 + 底线；subsection 左小色块） ----------
\usepackage{titlesec}

% section：序号前加 4pt 蓝色竖条，主题色标题，下方细横线
\titleformat{\section}[hang]

    {\makebox[0pt][l]{\color{accent}\rule[-0.15em]{4pt}{1.15em}}\hspace{0.6em}\thesection}

    {\heiti\zihao{3}\color{accent!85!black}}

    {0.5em}{}

% subsubsection：朴素黑体

    {\thesubsubsection}{0.5em}{}
