% ---------- 页面 ----------
\usepackage[a4paper, top=2.5cm, bottom=2.5cm, left=2.5cm, right=2.5cm]{geometry}

% ---------- 行距与段距 ----------
% 国赛通用：1.3 倍行距，段间距清零，首行缩进两字符
\linespread{1.3}
\setlength{\parskip}{0pt}
\setlength{\parindent}{2em}

% ---------- 英文主字体（Times New Roman，国赛惯用） ----------
% Windows 自带 Times New Roman；如为其他系统，xelatex 会自动回退到默认 serif
\usepackage{fontspec}
\IfFontExistsTF{Times New Roman}{\setmainfont{Times New Roman}}{}

% ---------- 数学 ----------
\usepackage{amsmath, amssymb, amsthm, mathtools, bm}
\numberwithin{equation}{section}

% ---------- 单位 / 数字 ----------
\usepackage{siunitx}
\sisetup{detect-all, per-mode=symbol, inter-unit-product=\ensuremath{{}\cdot{}}}

% ---------- 图表 ----------
\usepackage{graphicx}
\usepackage{booktabs}
\usepackage{array}
\usepackage{longtable}
\usepackage{multirow}
\usepackage{caption}
\usepackage{subcaption}
\usepackage{float}
\captionsetup{font=small, labelfont=bf, skip=4pt}
\numberwithin{figure}{section}
\numberwithin{table}{section}
\graphicspath{{figures/}}

% ---------- TikZ 流程图 ----------
\usepackage{tikz}
\usetikzlibrary{arrows.meta, shapes.geometric, positioning, calc}

% ---------- 列表（紧凑） ----------
\usepackage{enumitem}
\setlist{itemsep=3pt, topsep=3pt, parsep=0pt}

% ---------- 算法伪代码 ----------
\usepackage[ruled, linesnumbered]{algorithm2e}
\SetAlgorithmName{算法}{算法}{算法列表}

% ---------- 链接（电子版建议蓝色，打印版可改为 black） ----------
\usepackage{url}
\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    linkcolor=blue!60!black,
    citecolor=blue!60!black,
    urlcolor=blue,
    bookmarksnumbered=true
}

% ---------- 代码块（保留原始缩进，等宽对齐） ----------
\usepackage{xcolor}
\usepackage{listings}
\definecolor{codegray}{rgb}{0.5,0.5,0.5}
\definecolor{codebg}{rgb}{0.97,0.97,0.97}
\definecolor{codekw}{rgb}{0.00,0.00,0.70}
\definecolor{codestr}{rgb}{0.58,0.00,0.00}
\definecolor{codecom}{rgb}{0.25,0.50,0.35}
\lstset{
    backgroundcolor=\color{codebg},
    basicstyle=\ttfamily\footnotesize,
    keywordstyle=\color{codekw}\bfseries,
    stringstyle=\color{codestr},
    commentstyle=\color{codecom}\itshape,
    numbers=left,
    numberstyle=\tiny\color{codegray},
    frame=single,
    rulecolor=\color{codegray},
    breaklines=true,
    breakatwhitespace=true,
    showstringspaces=false,
    tabsize=4,
    captionpos=b,
    columns=fixed,
    keepspaces=true,
    extendedchars=false,
    inputencoding=utf8
}

% ---------- 参考文献（国标 GB/T 7714-2015） ----------
\usepackage[backend=biber, style=gb7714-2015, gbpub=false]{biblatex}
\addbibresource{refs.bib}

% ---------- 定理环境 ----------
\newtheorem{theorem}{定理}[section]
\newtheorem{lemma}[theorem]{引理}
\newtheorem{proposition}[theorem]{命题}
\newtheorem{corollary}[theorem]{推论}
\theoremstyle{definition}
\newtheorem{definition}{定义}[section]
\newtheorem{example}{例}[section]
\newtheorem{remark}{注}[section]

% ---------- 章节标题（xelatex + ctex 陷阱：heiti 已是粗体，勿再叠加 bfseries） ----------
\usepackage{titlesec}
\titleformat{\section}{\heiti\zihao{-3}}{\thesection}{1em}{}
\titleformat{\subsection}{\heiti\zihao{4}}{\thesubsection}{1em}{}
\titleformat{\subsubsection}{\heiti\zihao{-4}}{\thesubsubsection}{1em}{}
\titlespacing*{\section}{0pt}{12pt}{6pt}
\titlespacing*{\subsection}{0pt}{8pt}{4pt}
\titlespacing*{\subsubsection}{0pt}{6pt}{3pt}

% ---------- 目录（压缩间距，限定到 section，简洁居中标题） ----------
\setcounter{tocdepth}{1}
\renewcommand{\contentsname}{\centerline{\heiti\zihao{-2} 目\quad 录}}
\usepackage{titletoc}
\titlecontents{section}[2em]{\vspace{2pt}\heiti}
    {\contentslabel{2em}}{\hspace*{-2em}}
    {\titlerule*[8pt]{.}\contentspage}

% ---------- 页眉页脚 ----------
\usepackage{fancyhdr}
\setlength{\headheight}{14pt}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small 2026 西邮校赛数学建模}
\fancyhead[R]{\small \leftmark}
\fancyfoot[C]{\small \thepage}
\renewcommand{\headrulewidth}{0.4pt}

% 摘要页使用 plain 样式，保留页码（罗马数字 I）
\fancypagestyle{abstractstyle}{%
    \fancyhf{}
    \fancyfoot[C]{\small \thepage}
    \renewcommand{\headrulewidth}{0pt}
}

% ---------- 自定义命令 ----------
% 摘要页标题（居中"摘 要"黑体小二）
\newcommand{\abstracttitle}{%
    \begin{center}
        {\heiti\zihao{-2} 摘\quad 要}
    \end{center}
    \vspace{0.5em}
}

% 关键词（国赛惯例：分号分隔）
\newcommand{\keywords}[1]{%
    \vspace{1em}
    \noindent{\heiti 关键词：}\quad #1
}

% 封面用下划线填充框（对齐基线，不随字体漂浮）
\newcommand{\ulfield}[1]{\underline{\makebox[#1][c]{\ }}}
