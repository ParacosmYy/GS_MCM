% ---------- 章节标题（全黑，靠字号和装饰线区分层级） ----------
\usepackage{titlesec}

% section：小二黑体 + 底部 0.8pt 横线（单线，避免双 \rule{\linewidth} 触发 overfull）
\titleformat{\section}[hang]
    {\heiti\zihao{-2}}
    {\thesection}{0.8em}{}
    [{\vspace{-4pt}\hrule height 0.8pt}]

% subsection：三号黑体（极简，无装饰）
\titleformat{\subsection}[hang]
    {\heiti\zihao{3}}
    {\thesubsection}{0.8em}{}

% subsubsection：四号黑体
\titleformat{\subsubsection}[hang]
    {\heiti\zihao{4}}
    {\thesubsubsection}{0.5em}{}
