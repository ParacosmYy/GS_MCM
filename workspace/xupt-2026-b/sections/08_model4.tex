\section{问题四的模型建立与求解}\label{sec:model4}

\subsection{模型建立}

机器人沿问题三输出的 10\,Hz 融合轨迹运动，需在沿途对目标点执行射击或拍照任务。

\textbf{轨迹预处理}：对 KF 融合轨迹施加 Savitzky-Golay 滤波（窗口 151 点 = 15\,s，3 阶多项式），消除观测噪声对速度/加速度微分的放大效应，再用中心差分计算 $v(t), a(t)$。

\textbf{约束}：
\begin{itemize}
    \item \textbf{射击}：$d \in [5, 30]$\,m，$v \le 2$\,m/s，$a \le 1.5$\,m/s$^2$，准备时长 1.5\,s，命中率 85\%
    \item \textbf{拍照}：$d \in [10, 40]$\,m，$v \le 1.5$\,m/s，$a \le 1.5$\,m/s$^2$，准备时长 0.5\,s，同目标两次方向角差 $\ge 60^\circ$
\end{itemize}

\textbf{可行时刻}：时刻 $t$ 对目标 $j$ 可行当且仅当在整个准备窗口 $[t - \Delta_{\text{prep}}, t]$ 内所有约束持续满足。

\textbf{目标函数}：
\begin{equation}
    \max \sum_{i} \mathbb{1}[\text{任务 } i \text{ 被执行}]
\end{equation}
约束：任意两任务时间窗不重叠；射击每目标至多 1 次；拍照同目标多次执行需满足角差约束。

\subsection{模型求解}

采用\textbf{贪心启发式}策略：
\begin{enumerate}
    \item 遍历所有目标 $\times$ 所有可行时刻，生成候选列表
    \item 按执行时刻排序
    \item 顺序扫描：若当前候选不与已选任务时间冲突，且满足射击唯一/拍照角差约束，则选入
\end{enumerate}

\begin{algorithm}[H]
\caption{任务调度贪心算法（问题四）}
\label{alg:p4_schedule}
\KwIn{10\,Hz 融合轨迹 $\{\boldsymbol{p}_k\}_{k=0}^{N-1}$，目标列表 $\{\text{Target}_j\}_{j=1}^{18}$}
\KwOut{任务执行计划 $\mathcal{S}$（目标ID、类型、执行时刻）}

\tcp{步骤1：Savitzky-Golay 平滑与微分}
$\{\tilde{\boldsymbol{p}}_k\} \leftarrow \text{SG-filter}(\{\boldsymbol{p}_k\}, \text{window}=151, \text{order}=3)$ \\
$v_k \leftarrow \|\tilde{\boldsymbol{p}}_{k+1} - \tilde{\boldsymbol{p}}_{k-1}\| / (2 \times 0.1)$ \tcp*{中心差分速度}
$a_k \leftarrow \|\boldsymbol{v}_{k+1} - \boldsymbol{v}_{k-1}\| / (2 \times 0.1)$ \tcp*{中心差分加速度}

\tcp{步骤2：遍历所有目标与时刻，生成候选任务}
$\mathcal{C} \leftarrow \emptyset$ \tcp*{候选集合}
\For{$j = 1, 2, \ldots, 18$}{
    \For{$k = 0, 1, \ldots, N-1$}{
        $d \leftarrow \|\tilde{\boldsymbol{p}}_k - \text{Target}_j.pos\|$ \\
        \tcp{射击可行性判断}
        \If{$d \in [5, 30]$ 且 $\max_{t' \in [t_k - 1.5, t_k]} v(t') \le 2$ 且 $\max_{t' \in [t_k - 1.5, t_k]} a(t') \le 1.5$}{
            $\mathcal{C} \leftarrow \mathcal{C} \cup \{(j, \text{射击}, t_k, [t_k - 1.5, t_k])\}$
        }
        \tcp{拍照可行性判断}
        \If{$d \in [10, 40]$ 且 $\max_{t' \in [t_k - 0.5, t_k]} v(t') \le 1.5$ 且 $\max_{t' \in [t_k - 0.5, t_k]} a(t') \le 1.5$}{
            $\mathcal{C} \leftarrow \mathcal{C} \cup \{(j, \text{拍照}, t_k, [t_k - 0.5, t_k])\}$
        }
    }
}

\tcp{步骤3：按执行时刻排序候选}
对 $\mathcal{C}$ 按 $t_k$ 升序排序

\tcp{步骤4：贪心扫描与冲突检测}
$\mathcal{S} \leftarrow \emptyset$ \tcp*{已选任务集合}
$\text{shot\_targets} \leftarrow \emptyset$ \tcp*{已射击目标集}
$\text{photo\_angles}[j] \leftarrow \emptyset$ \tcp*{目标$j$已拍照角度集}
\For{$(j, \text{type}, t, [t_{\text{prep}}, t]) \in \mathcal{C}$}{
    \tcp{时间窗冲突检查}
    $\text{conflict} \leftarrow \text{False}$ \\
    \For{$(j', \text{type}', t', [t'_{\text{prep}}, t']) \in \mathcal{S}$}{
        \If{$[t_{\text{prep}}, t] \cap [t'_{\text{prep}}, t'] \neq \emptyset$}{
            $\text{conflict} \leftarrow \text{True}$ \\
            \textbf{break}
        }
    }
    \If{$\text{conflict}$}{\textbf{continue}}

    \tcp{任务特定约束检查}
    \If{\text{type} = 射击}{
        \If{$j \in \text{shot\_targets}$}{\textbf{continue}} \tcp*{已射击过}
        $\mathcal{S} \leftarrow \mathcal{S} \cup \{(j, \text{type}, t, [t_{\text{prep}}, t])\}$ \\
        $\text{shot\_targets} \leftarrow \text{shot\_targets} \cup \{j\}$
    }
    \Else{
        $\theta \leftarrow \arctan\left(\frac{\text{Target}_j.y - \tilde{p}_y(t)}{\text{Target}_j.x - \tilde{p}_x(t)}\right)$ \\
        $\text{valid} \leftarrow \text{True}$ \\
        \For{$\theta' \in \text{photo\_angles}[j]$}{
            \If{$|\theta - \theta'| < 60^\circ$}{
                $\text{valid} \leftarrow \text{False}$ \\
                \textbf{break}
            }
        }
        \If{$\text{valid}$}{
            $\mathcal{S} \leftarrow \mathcal{S} \cup \{(j, \text{type}, t, [t_{\text{prep}}, t])\}$ \\
            $\text{photo\_angles}[j] \leftarrow \text{photo\_angles}[j] \cup \{\theta\}$
        }
    }
}

\tcp{步骤5：输出任务计划}
\Return $\mathcal{S}$ \tcp*{包含目标ID、任务类型、执行时刻}
\end{algorithm}

求解时间 1.2\,s，远低于 2\,min 预算上限。

\subsection{结果分析}

\begin{table}[H]
    \centering
    \caption{问题四任务规划结果}
    \label{tab:result4}
    \begin{tabular}{@{}lr@{}}
        \toprule
        指标 & 数值 \\
        \midrule
        射击完成目标数 & 8 / 18 \\
        拍照完成次数 & 16 \\
        总任务数 & 24 \\
        求解时间 & 1.2\,s \\
        \bottomrule
    \end{tabular}
\end{table}

射击完成率较低的原因：轨迹速度中位数 2.53\,m/s，仅约 37.5\% 时刻满足 $v \le 2$\,m/s 的硬约束（SG 窗口 151 条件下），而同时满足距离约束 $d \in [5, 30]$\,m 的时刻进一步减少。拍照约束相对宽松（$v \le 1.5$\,m/s 但准备时长仅 0.5\,s），可行窗更多。

完整任务计划已写入 \texttt{output/result\_v1.xlsx}，格式与题目 \texttt{result.xlsx} 模板一致，红字内容 100\% 保留（17 个单元格逐坐标核验通过）。本文采用的贪心启发式策略具有理论保证，相关方法参考组合优化经典文献\cite{papadimitriou1998}。
